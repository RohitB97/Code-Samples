<button type=button onclick="adds()" >Add</button>
</form>
<p id="list"></p>

<script>
var text=new Array;      

function display()   <!--displays the list after adding, removing, and changing a task-->
{ var k,i;
  k="<ol>";
  for(i=0;i<text.length;i++)
   { k+="<li>"+ text[i] +" "+"<button id='" + i + "' onclick='cut(this)'>X</button>"+" "+"<button id='" + i + "' onclick='change(this)'>change</button>"+"</li>"; }
   k+="</ol>";
 document.getElementById("list").innerHTML=k;
}

function adds()    <!--adds a task to the list-->
{  if(todo.input.value!='')
    {text.push(todo.input.value);}
   display();
}

function cut(e)  <!--removes a task from the list-->
{ var x= e.getAttribute("id");
  text.splice(x,1);
  display();
}

function change(e)  <!--changes that particular task-->
{ var y= e.getAttribute("id");
  if(todo.input.value!='')
   { text[y]=todo.input.value; }
  display();
}

function blank() <!--empties the input box when it is focused-->
{ todo.input.value="";
}

</script>
